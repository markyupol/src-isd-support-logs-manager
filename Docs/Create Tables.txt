
CREATE TABLE SUPP (
       SUPP_ID NUMBER,
       SEC_ID NUMBER,
       DEPT_ID NUMBER,
       SHIFT_ID NUMBER,
       CATG_ID NUMBER,
       SCOP_ID NUMBER,
       START_DATETIME DATE,
       END_DATETIME DATE,
       SUBJECT VARCHAR2(100),
       DETAILS VARCHAR2(500),
       CAUSE VARCHAR2(500),
       ACTION_TAKEN VARCHAR2(500),
       ATTENDED_BY_ACCT_ID NUMBER,
       POSTED_DATETIME DATE,
       POSTED_BY_ACCT_ID NUMBER
) TABLESPACE TABLEAU_USER;
